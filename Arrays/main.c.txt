#include <stdio.h>

int main()
{
    int a[5], i, sum=0, j=1, n;
    printf("Enter the size of the array: ");
    scanf("%d", &n);
    
    for(i=0; i<n; i++){
        printf("Elements %d: ", j++);
        scanf("%d", &a[i]);
    }
    
    for(i=0; i<n; i++){
        sum+=a[i];
    }
    
    printf("Sum of all array elements: %d", sum);
    return 0;
}
